00107         void calculate_derived_params(bool report = false)
00108         {
00109                 using namespace std;
00110                 
00111                 dp = 2.*M_PI/len;
00112 
00113                 // We assume that either a mass term or a lambda term dominates.
00114                 R phi2 = m_phi/2.0 * pow<2>(phi0);
00115                 R chi2 = m_chi/2.0 * pow<2>(chi0);
00116 
00117                 R lambda_phi_eff = lambda_phi;
00118                 R lambda_chi_eff = lambda_chi;
00119                 R gamma_phi_eff = gamma_phi;
00120                 R gamma_chi_eff = gamma_chi;
00121 
00122                 // Monodromy potential: c s^2 ([ 1 + (phi/s)^2 ]^e - 1)
00123                 // = c ( e phi^2 + (1/2) s^{-2} (e^2 - e) phi^4 + (1/6) s^{-4} (e^3 - 3 e^2 + 2 e) phi^6 + ... )
00124                 if (md_e_phi != 0) {
00125                         md_c_phi = 0.5*pow<2>(m_phi)/md_e_phi;
00126 
00127                         lambda_phi_eff += 4.0*md_c_phi*0.5*md_e_phi*(md_e_phi - 1)/pow<2>(md_s_phi);
00128                         gamma_phi_eff += 6.0*md_c_phi*md_e_phi*(md_e_phi*(md_e_phi - 3) + 2)/(6.0*pow<4>(md_s_phi));
00129                 }
00130                 if (md_e_chi != 0) {
00131                         md_c_chi = 0.5*pow<2>(m_chi)/md_e_chi;
00132 
00133                         lambda_chi_eff += 4.0*md_c_chi*0.5*md_e_chi*(md_e_chi - 1)/pow<2>(md_s_chi);
00134                         gamma_chi_eff += 6.0*md_c_chi*md_e_chi*(md_e_chi*(md_e_chi - 3) + 2)/(6.0*pow<4>(md_s_chi));
00135                 }
00136 
00137 #ifdef FULL_RESCALE
00138                 R phi4 = lambda_phi_eff/4.0 * pow<4>(phi0);
00139                 R chi4 = lambda_chi_eff/4.0 * pow<4>(chi0);
00140                 R phi6 = gamma_phi_eff/6.0 * pow<6>(phi0);
00141                 R chi6 = gamma_chi_eff/6.0 * pow<6>(chi0);
00142 #endif
00143 
00144                 R phi2mag = fabs(phi2);
00145                 R chi2mag = fabs(chi2);
00146 #ifdef FULL_RESCALE
00147                 R phi4mag = fabs(phi4);
00148                 R chi4mag = fabs(chi4);
00149                 R phi6mag = fabs(phi6);
00150                 R chi6mag = fabs(chi6);
00151 #endif
00152 
00153                 std::set<rs_init<R> > rss;
00154                 if (phi0 != 0) rss.insert(rs_init<R>(phi2mag, m_phi, 0, 1.5, 1/phi0, "phi^2"));
00155                 if (chi0 != 0) rss.insert(rs_init<R>(chi2mag, m_chi, 0, 1.5, 1/chi0, "chi^2"));
00156 #ifdef FULL_RESCALE
00157                 if (phi0 != 0) rss.insert(rs_init<R>(phi4mag, phi0 * sqrt(fabs(lambda_phi_eff)), -1, 1, 1/phi0, "phi^4"));
00158                 if (chi0 != 0) rss.insert(rs_init<R>(chi4mag, chi0 * sqrt(fabs(lambda_chi_eff)), -1, 1, 1/chi0, "chi^4"));
00159                 if (phi0 != 0) rss.insert(rs_init<R>(phi6mag, pow<2>(phi0) * sqrt(fabs(gamma_phi_eff)), -3./2., 3./4., 1/phi0, "phi^6"));
00160                 if (chi0 != 0) rss.insert(rs_init<R>(chi6mag, pow<2>(chi0) * sqrt(fabs(gamma_chi_eff)), -3./2., 3./4., 1/chi0, "chi^6"));
00161 #endif
00162 
00163                 if (rss.size()) {
00164                         rescale_B = rss.begin()->rescale_B;
00165                         rescale_s = rss.begin()->rescale_s;
00166                         rescale_r = rss.begin()->rescale_r;
00167                         rescale_A = rss.begin()->rescale_A;
00168                 }
00169                 else {
00170                         rescale_B = 1.0;
00171                         rescale_s = 0;
00172                         rescale_r = 0;
00173                         rescale_A = 1.0;
00174                 }
00175 
00176                 if (report) std::cout << "Initially Dominant Term: " << rss.begin()->desc << std::endl;
00177         }

00185         R V(R phi, R chi, R a_t)
00186         {
00187                 using namespace std;
00188 
00189                 const R tophys = 1./rescale_A * pow(a_t, -rescale_r);
00190                 const R phi_phys = tophys * phi;
00191                 const R chi_phys = tophys * chi;
00192                 return pow<2>(rescale_A / rescale_B) * pow(a_t, -2. * rescale_s + 2. * rescale_r) *
00193                         (
00194                                 (
00195                                         (md_e_phi != 0) ?
00196                                         md_c_phi*pow<2>(md_s_phi)*(pow(1.0 + pow<2>(phi_phys/md_s_phi), md_e_phi) - 1.0) :
00197                                         0.5*pow<2>(m_phi * phi_phys)
00198                                 ) +
00199                                 (
00200                                         (md_e_chi != 0) ?
00201                                         md_c_chi*pow<2>(md_s_chi)*(pow(1.0 + pow<2>(chi_phys/md_s_chi), md_e_chi) - 1.0) :
00202                                         0.5*pow<2>(m_chi * chi_phys)
00203                                 ) +
00204                                 0.25*lambda_phi*pow<4>(phi_phys) +
00205                                 0.25*lambda_chi*pow<4>(chi_phys) +
00206                                 0.5*pow<2>(g * phi_phys * chi_phys) +
00207                                 gamma_phi*pow<6>(phi_phys)/6.0 +
00208                                 gamma_chi*pow<6>(chi_phys)/6.0
00209                         );

00293                 using namespace std;

00295                 if (pwr_exp) {
