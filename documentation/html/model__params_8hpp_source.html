00097         void calculate_derived_params(bool report = false)
00098         {
00099                 using namespace std;
00100                 
00101                 dp = 2.*M_PI/len;
00102 
00103                 // We assume that either a mass term or a lambda term dominates.
00104                 R phi2 = m_phi/2.0 * pow<2>(phi0);
00105                 R chi2 = m_chi/2.0 * pow<2>(chi0);
00106                 R phi4 = lambda_phi/4.0 * pow<4>(phi0);
00107                 R chi4 = lambda_chi/4.0 * pow<4>(chi0);
00108                 R phi6 = gamma_phi/6.0 * pow<6>(phi0);
00109                 R chi6 = gamma_chi/6.0 * pow<6>(chi0);
00110 
00111                 R phi2mag = fabs(phi2);
00112                 R chi2mag = fabs(chi2);
00113                 R phi4mag = fabs(phi4);
00114                 R chi4mag = fabs(chi4);
00115                 R phi6mag = fabs(phi6);
00116                 R chi6mag = fabs(chi6);
00117 
00118                 std::set<rs_init<R> > rss;
00119                 if (phi0 != 0) rss.insert(rs_init<R>(phi2mag, m_phi, 0, 1.5, 1/phi0, "phi^2"));
00120                 if (chi0 != 0) rss.insert(rs_init<R>(chi2mag, m_chi, 0, 1.5, 1/chi0, "chi^2"));
00121                 if (phi0 != 0) rss.insert(rs_init<R>(phi4mag, phi0 * sqrt(fabs(lambda_phi)), -1, 1, 1/phi0, "phi^4"));
00122                 if (chi0 != 0) rss.insert(rs_init<R>(chi4mag, chi0 * sqrt(fabs(lambda_chi)), -1, 1, 1/chi0, "chi^4"));
00123                 if (phi0 != 0) rss.insert(rs_init<R>(phi6mag, pow<2>(phi0) * sqrt(fabs(gamma_phi)), -3./2., 3./4., 1/phi0, "phi^6"));
00124                 if (chi0 != 0) rss.insert(rs_init<R>(chi6mag, pow<2>(chi0) * sqrt(fabs(gamma_chi)), -3./2., 3./4., 1/chi0, "chi^6"));
00125 
00126                 if (rss.size()) {
00127                         rescale_B = rss.begin()->rescale_B;
00128                         rescale_s = rss.begin()->rescale_s;
00129                         rescale_r = rss.begin()->rescale_r;
00130                         rescale_A = rss.begin()->rescale_A;
00131                 }
00132                 else {
00133                         rescale_B = 1.0;
00134                         rescale_s = 0;
00135                         rescale_r = 0;
00136                         rescale_A = 1.0;
00137                 }
00138 
00139                 if (report) std::cout << "Initially Dominant Term: " << rss.begin()->desc << std::endl;
00140         }

00148         R V(R phi, R chi, R a_t)
00149         {
00150                 using namespace std;
00151 
00152                 const R tophys = 1./rescale_A * pow(a_t, -rescale_r);
00153                 const R phi_phys = tophys * phi;
00154                 const R chi_phys = tophys * chi;
00155                 return pow<2>(rescale_A / rescale_B) * pow(a_t, -2. * rescale_s + 2. * rescale_r) *
00156                         (
00157                                 0.5*pow<2>(m_phi * phi_phys) +
00158                                 0.5*pow<2>(m_chi * chi_phys) +
00159                                 0.25*lambda_phi*pow<4>(phi_phys) +
00160                                 0.25*lambda_chi*pow<4>(chi_phys) +
00161                                 0.5*pow<2>(g * phi_phys * chi_phys) +
00162                                 gamma_phi*pow<6>(phi_phys)/6.0 +
00163                                 gamma_chi*pow<6>(chi_phys)/6.0
00164                         );
